\begin{itemize}
    \item Respostas sem justificativas \textbf{NÃO} serão consideradas.
    \item Essa prova tem o valor de \textbf{19 pontos}.
    \item Todos os itens das questões $1-3$ valem \textbf{2 pontos}. A questão 4 vale \textbf{3 pontos}.
\end{itemize}

\begin{enumerate}
    \item Dado o PVI
    \[
    \begin{cases}
        \dfrac{dy}{dx}=\dfrac{2x+1}{3y^2-3},\\
        y(0)=0,
    \end{cases}
    \]
    faça o que se pede:
    \begin{enumerate}
        \item Obtenha a solução do PVI.
        \item Determine o intervalo de validade da solução.
        \item Determine os pontos onde a solução tem um máximo local.
    \end{enumerate}
    \newpage
    \item Considere a seguinte equação diferencial
    \[2y^2 + \frac{2y}{x} + \left(2xy+2+\frac{y}{x}\right)y' = 0.\]
    \begin{enumerate}
        \item Mostre que a equação diferencial não é exata e que um fator integrante é $\mu(x)=x$.
        \item Obtenha a solução geral da equação.
        \item Encontre a solução que satisfaz $y(1)=1$.
    \end{enumerate}
    \newpage
    \item Para a equação diferencial
    \[\dfrac{dy}{dt} = 1-y^2\]
    faça o que se pede:
    \begin{enumerate}
        \item Esboce o gráfico de $f(y)$ em função de $y$, determine os pontos de equilíbrio e classifique cada um dos pontos de equilíbrio como assintoticamente estável ou instável.
        \item Determine para quais valores de $y$ as soluções têm pontos de inflexão.
        \item Faça um esboço das soluções levando em conta os itens $(a)$ e $(b)$.

    \end{enumerate}
    \newpage
    \item Considere o PVI a seguir
    \[
    \begin{cases}
        (t^2-t)\dfrac{dy}{dt}+(t+1)y=e^t,\\\\
        y(t_0)=y_0.
    \end{cases}
    \]
    Determine o intervalo de validade da solução do PVI, sem resolvê-lo, em função de $(t_0,y_0)$.
\end{enumerate}
